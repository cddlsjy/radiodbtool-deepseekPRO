二、同步逻辑的“无 bug”实现规范（Trae 必须遵守）
您反馈 deepseekPRO 可能无法下载全部数据库，原因很可能在于全量同步时没有正确使用偏移量或没有处理服务器返回空数据的情况。下面是经过验证的正确写法：

kotlin
// SyncManager.kt - 全量同步（保证下载全部）
suspend fun syncFull(onProgress: (Int, Int, String) -> Unit) {
    // 1. 获取总电台数
    val total = repository.fetchStats()
    if (total == 0) {
        onProgress(0, 0, "无法获取电台总数")
        return
    }
    val filterKey = "full"
    var offset = db.syncProgressDao().getOffset(filterKey) ?: 0
    if (offset >= total) {
        onProgress(total, total, "数据已是最新")
        db.syncProgressDao().deleteProgress(filterKey)
        return
    }
    // 2. 分页下载，每页 500 条（与 Python 脚本一致）
    while (offset < total && isActive) {
        val stations = repository.fetchStationPage(offset = offset, limit = 500)
        if (stations.isEmpty()) break
        repository.insertStations(stations)
        offset += stations.size
        db.syncProgressDao().saveProgress(SyncProgressEntity(filterKey, offset))
        onProgress(offset, total, "已下载 $offset / $total")
        // 避免请求过快
        delay(50)
    }
    // 3. 完成后清除断点
    db.syncProgressDao().deleteProgress(filterKey)
    onProgress(total, total, "全量同步完成")
}
关键点：

偏移量 offset 每次增加实际获取的电台数量，而非固定 500（避免服务器返回不足 500 时越界）

每页大小 limit = 500 与 Python 版本保持一致，避免服务器限制

使用 while (offset < total && isActive) 确保取消和边界正确

最后必须清除 sync_progress 表中的记录，否则下次同步会认为已完成

fun toM3U(stations: List<StationEntity>): String {
    return buildString {
        append("#EXTM3U\n\n")
        stations.forEach { station ->
            if (!station.url.isNullOrEmpty()) {
                append("#EXTINF:-1,${station.name?.replace("\n", "") ?: "Unknown"}\n")
                station.favicon?.let { append("#EXTIMG:$it\n") }
                append("${station.url}\n")
                append("#RADIOBROWSERUUID:${station.stationuuid}\n\n")
            }
        }
    }
}
2. CSV 格式（正确转义逗号和双引号）
kotlin
fun toCSV(stations: List<StationEntity>): String {
    val headers = listOf("stationuuid", "name", "url", "homepage", "favicon", "country", "countrycode", "state", "language", "tags", "codec", "bitrate", "lastcheckok", "lastchangetime", "clickcount", "clicktrend", "votes")
    val out = StringBuilder()
    out.append(headers.joinToString(",")).append("\n")
    for (s in stations) {
        val row = headers.map { header ->
            val value = when (header) {
                "stationuuid" -> s.stationuuid
                "name" -> s.name
                "url" -> s.url
                "homepage" -> s.homepage
                "favicon" -> s.favicon
                "country" -> s.country
                "countrycode" -> s.countrycode
                "state" -> s.state
                "language" -> s.language
                "tags" -> s.tags
                "codec" -> s.codec
                "bitrate" -> s.bitrate?.toString()
                "lastcheckok" -> s.lastcheckok?.toString()
                "lastchangetime" -> s.lastchangetime
                "clickcount" -> s.clickcount?.toString()
                "clicktrend" -> s.clicktrend?.toString()
                "votes" -> s.votes?.toString()
                else -> ""
            }
            escapeCsv(value)
        }
        out.append(row.joinToString(",")).append("\n")
    }
    return out.toString()
}

private fun escapeCsv(value: String?): String {
    if (value.isNullOrEmpty()) return ""
    // 如果包含逗号、换行或双引号，则用双引号包裹并转义内部双引号
    return if (value.contains(",") || value.contains("\n") || value.contains("\"")) {
        "\"" + value.replace("\"", "\"\"") + "\""
    } else value
}

4. SQLite DB 导出（使用 Room 临时数据库，避免损坏主库）
kotlin
suspend fun exportToDatabase(stations: List<StationEntity>, context: Context, uri: Uri) {
    withContext(Dispatchers.IO) {
        val tempFile = File(context.cacheDir, "temp_export_${System.currentTimeMillis()}.db")
        val tempDb = Room.databaseBuilder(context, RadioDatabase::class.java, tempFile.absolutePath)
            .build()
        // 批量插入
        tempDb.stationDao().insertStations(stations)
        tempDb.close()
        // 复制到目标 URI
        context.contentResolver.openOutputStream(uri)?.use { output ->
            FileInputStream(tempFile).use { input ->
                input.copyTo(output)
            }
        }
        tempFile.delete()
    }
}

fun startSync() {
    viewModelScope.launch {
        _uiState.update { it.copy(isSyncing = true) }
        try {
            if (_uiState.value.syncMode == SyncMode.FULL) {
                syncManager.syncFull { current, total, msg ->
                    _uiState.update { state ->
                        state.copy(
                            syncProgress = current.toFloat() / total,
                            syncMessage = msg
                        )
                    }
                }
            } else {
                // 条件同步
                syncManager.syncFiltered(/*...*/)
            }
            // 刷新本地列表
            refreshLocalData()
        } catch (e: CancellationException) {
            _uiState.update { it.copy(syncMessage = "已取消") }
        } finally {
            _uiState.update { it.copy(isSyncing = false) }
        }
    }
